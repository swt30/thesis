\makeatletter
  \def\uc@dclc#1#2#3{%
    \ifnum\pdfstrcmp{#2}{mathletters}=\z@
      \begingroup\edef\x{\endgroup
        \noexpand\DeclareUnicodeCharacter{\hex{#1}}}\x{#3}%
    \fi
  }
  \input{uni-3.def}
  \def\uc@dclc#1#2#3{%
    \ifnum\pdfstrcmp{#2}{default}=\z@
      \begingroup\edef\x{\endgroup
        \noexpand\DeclareUnicodeCharacter{\hex{#1}}}\x{#3}%
    \fi
  }
  \input{uni-34.def}
\makeatother

  \end{fullwidth}%

% Contents page layout

  {\hspace*{2em}\contentslabel{2em}} % before w/label (label = ``2'')

  [\end{fullwidth}] % after
